21  package org.synchronoss.cpo.jdbc;
22  
23  import java.io.Serializable;
24  
25  public class LobValueObjectBean implements LobValueObject, Serializable {
26  
27    /**
28     * Version Id for this class.
29     */
30    private static final long serialVersionUID = 1L;
31    private int lobId;
32    private byte[] bLob = null;
33    private byte[] bLob2 = null;
34    private char[] cLob = null;
35  
36    public LobValueObjectBean() {
37    }
38  
39    public LobValueObjectBean(int id, byte[] bLob, char[] cLob) {
40      this.lobId = id;
41      this.bLob = bLob;
42      this.cLob = cLob;
43    }
44  
45    public void setLobId(int lobId) {
46      this.lobId = lobId;
47    }
48  
49    public int getLobId() {
50      return this.lobId;
51    }
52  
53    public void setBLob(byte[] bLob) {
54      this.bLob = bLob;
55    }
56  
57    public byte[] getBLob() {
58      return this.bLob;
59    }
60  
61    public void setBLob2(byte[] bLob) {
62      this.bLob2 = bLob;
63    }
64  
65    public byte[] getBLob2() {
66      return this.bLob2;
67    }
68  
69    public void setCLob(char[] cLob) {
70      this.cLob = cLob;
71    }
72  
73    public char[] getCLob() {
74      return this.cLob;
75    }
76  }
