25  import org.apache.jmeter.samplers.SampleResult;

28  import org.synchronoss.cpo.jdbc.ValueObject;

30  import org.synchronoss.cpo.jdbc.meta.JdbcCpoMetaDescriptor;
31  import org.synchronoss.cpo.meta.CpoMetaDescriptor;

33  import java.sql.Timestamp;

35  /**
36   * JMeter Samplier Client to allow a load test of cpo
37   *
38   * @author Michael Bellomo
39   * @since 5/9/12
40   */
41  public class CpoJavaSamplerClient extends AbstractJavaSamplerClient {
42  
43    private Logger logger = getLogger();
44  
45    private JdbcCpoMetaDescriptor metaDescriptor = null;
46    private CpoAdapter cpoAdapter = null;
47  
48    private static final String START_ID = "Starting Id";
49    private static final String USER_NAME = "username";
50    private static final String PASSWORD = "password";
51    private static final String URL = "url";
52    private static final String DRIVER = "driver";
53    private static final String CONFIG_PROCESSOR = "org.synchronoss.cpo.jdbc.config.JdbcCpoConfigProcessor";
54  
55    @Override
56    public void setupTest(JavaSamplerContext javaSamplerContext) {
57      super.setupTest(javaSamplerContext);
58  
59      try {
60        metaDescriptor = (JdbcCpoMetaDescriptor)CpoMetaDescriptor.getInstance("jmeter-" + System.currentTimeMillis(), "/oracle/oracleValueMetaData.xml", true);
61  
62        CtJdbcConfig jdbcConfig = CtJdbcConfig.Factory.newInstance();
63        jdbcConfig.setName(this.getClass().getName());
64        jdbcConfig.setMetaDescriptorName(metaDescriptor.getName());
65        jdbcConfig.setCpoConfigProcessor(CONFIG_PROCESSOR);
66  
67        CtJdbcReadWriteConfig rwc = jdbcConfig.addNewReadWriteConfig();
68        rwc.setUser(javaSamplerContext.getParameter(USER_NAME));
69        rwc.setPassword(javaSamplerContext.getParameter(PASSWORD));
70        rwc.setUrl(javaSamplerContext.getParameter(URL));
71        rwc.setDataSourceClassName(javaSamplerContext.getParameter(DRIVER));
72  
73        cpoAdapter = CpoAdapterFactory.makeCpoAdapter(jdbcConfig);
74      } catch (Exception e) {
75        logger.error(e.getMessage());
76      }
77    }
78  
79    @Override
80    public void teardownTest(JavaSamplerContext javaSamplerContext) {
81      super.teardownTest(javaSamplerContext);
82      cpoAdapter = null;
83      metaDescriptor = null;
84    }
85  
86    @Override
87    public SampleResult runTest(JavaSamplerContext javaSamplerContext) {
88      SampleResult result = new SampleResult();
89      result.sampleStart();
90  
91      int id = Integer.parseInt(javaSamplerContext.getParameter(START_ID));
92  
93      ValueObject valueObject = new ValueObject(id);
94      result.setSampleLabel(Thread.currentThread().getName());
95  
96      valueObject.setAttrVarChar("testInsert");
97      valueObject.setAttrInteger(3);
98      Timestamp ts = new Timestamp(System.currentTimeMillis());
99  
100     if (!metaDescriptor.isSupportsMillis()) {
101       ts.setNanos(0);
102     }
103 
104     valueObject.setAttrDatetime(ts);
105     valueObject.setAttrBit(true);
106 
107     try {
108       cpoAdapter.insertObject(valueObject);
109       ValueObject vo = cpoAdapter.retrieveBean(null, valueObject, valueObject, null, null);
110       if (vo != null && vo.getId() == valueObject.getId()) {
111         result.setSuccessful(true);
112       } else {
113         result.setSuccessful(false);
114       }
115     } catch (CpoException ex) {
116       logger.error(ex.getMessage(), ex);
117       result.setSuccessful(false);
118     }
119     result.sampleEnd();
120 
121     return result;
122   }
123 
124   @Override
125   public Arguments getDefaultParameters() {
126     Arguments args = new Arguments();
127 
128     args.addArgument(new Argument(START_ID, "1"));
129     args.addArgument(new Argument(USER_NAME, "cpo"));
130     args.addArgument(new Argument(PASSWORD, "cpo"));
131     args.addArgument(new Argument(URL, ""));
132     args.addArgument(new Argument(DRIVER, ""));
133 
134     return args;
135   }
