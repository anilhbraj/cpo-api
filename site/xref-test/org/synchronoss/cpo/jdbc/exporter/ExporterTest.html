69    public void testXmlExport() {
70      logger.debug("testXmlExport");
71      try {
72        MetaXmlObjectExporter metaXmlObjectExporter = new JdbcMetaXmlObjectExporter(metaDescriptor);
73        for (CpoClass cpoClass : metaDescriptor.getCpoClasses()) {
74          cpoClass.acceptMetaDFVisitor(metaXmlObjectExporter);
75        }
76        CpoMetaDataDocument doc = metaXmlObjectExporter.getCpoMetaDataDocument();
77  
78        // doc better be valid
79        logger.debug("validating doc");
80        assertTrue(doc.validate());
81  
82        // make sure it saved the data right
83  
84        // should be 2 classes in here
85        assertEquals(2, doc.getCpoMetaData().getCpoClassArray().length);
86  
87        boolean found = false;
88        for (CtClass ctClass : doc.getCpoMetaData().getCpoClassArray()) {
89          // validate the ValueObject
90          if (ctClass.getName().equals(ValueObject.class.getName())) {
91            found = true;
92          }
93        }
94        assertTrue(found);
95      } catch (CpoException e) {
96        fail(e.getMessage());
97      }
98      logger.debug("testXmlExport complete");
99    }

101   public void testClassSourceExport() {
102     logger.debug("testClassSourceExport");
103     try {
104       CpoClassSourceGenerator classSourceGenerator = new CpoClassSourceGenerator(metaDescriptor);
105 
106       logger.debug("Generating java source");
107       CpoClass cpoClass = metaDescriptor.getMetaClass(new ValueObject());
108       cpoClass.acceptMetaDFVisitor(classSourceGenerator);
109       String classSource = classSourceGenerator.getSourceCode();
110 
111       // validate that we got something
112       assertNotNull(classSource);
113       assertFalse(classSource.isEmpty());
114 
115       // write the file
116       File javaFile = new File("target", ValueObject.class.getSimpleName() + ".java");
117       logger.debug("Saving class source to " + javaFile.getAbsolutePath());
118       FileWriter cw = new FileWriter(javaFile);
119       cw.write(classSourceGenerator.getSourceCode());
120       cw.flush();
121       cw.close();
122 
123       // let's try to compile the file
124       logger.debug("Compiling class source");
125       JavaCompiler compiler = ToolProvider.getSystemJavaCompiler();
126       StandardJavaFileManager fileManager = compiler.getStandardFileManager(null, null, null);
127       Iterable<? extends JavaFileObject> compilationUnits = fileManager.getJavaFileObjectsFromFiles(Arrays.asList(javaFile));
128       boolean result = compiler.getTask(null, fileManager, null, null, null, compilationUnits).call();
129 
130       // validate the result
131       assertTrue(result);
132 
133     } catch (Exception e) {
134       fail(e.getMessage());
135     }
136     logger.debug("testClassSourceExport complete");
137   }
