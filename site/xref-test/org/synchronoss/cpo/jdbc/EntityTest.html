21  package org.synchronoss.cpo.jdbc;
22  
23  import junit.framework.TestCase;
24  import org.slf4j.*;
25  import org.synchronoss.cpo.*;
26  import org.synchronoss.cpo.meta.domain.CpoAttribute;
27  
28  import java.util.List;
29  
30  /**
31   * ConstructorTest is a JUnit test class for testing the JdbcAdapter class Constructors
32   *
33   * @author david berry
34   */
35  public class EntityTest extends TestCase {
36  
37    private static final Logger logger = LoggerFactory.getLogger(EntityTest.class);
38  
39    public EntityTest(String name) {
40      super(name);
41    }
42  
43    /**
44     * <code>setUp</code> Load the datasource from the properties in the property file jdbc_en_US.properties
45     *
46     * @author david berry
47     * @version '$Id: ConstructorTest.java,v 1.7 2006/01/31 22:55:03 dberry Exp $'
48     */
49    @Override
50    public void setUp() {
51    }
52  
53    public void testGetDataSourceEntities() {
54      String method = "testGetDataSourceEntities:";
55      try {
56        CpoAdapter cpoAdapter = CpoAdapterFactory.getCpoAdapter(JdbcStatics.ADAPTER_CONTEXT_CLASS);
57        assertNotNull(method + "cpoAdapter is null", cpoAdapter);
58  
59        List<CpoAttribute> attributes = cpoAdapter.getCpoAttributes("select * from lob_test");
60        for (CpoAttribute attribute : attributes) {
61          if (!(attribute instanceof JdbcCpoAttribute))
62            fail("Attribute is not a JdbcCpoAttribute");
63          dumpAttribute((JdbcCpoAttribute)attribute);
64        }
65        assertTrue("List size is " + attributes, attributes.size() == 4);
66      } catch (Exception e) {
67        fail(method + e.getMessage());
68      }
69    }
70  
71    private void dumpAttribute(JdbcCpoAttribute attribute) {
72      logger.debug("DataName: "+attribute.getDataName());
73      logger.debug("DataType: "+attribute.getDataType());
74      logger.debug("DbColumn: "+attribute.getDbColumn());
75      logger.debug("DbTable: "+attribute.getDbTable());
76      logger.debug("JavaName: "+attribute.getJavaName());
77      logger.debug("JavaType: "+attribute.getJavaType());
78      logger.debug("DataTypeMapEntry: "+attribute.getDataTypeInt());
79    }
80  
81    @Override
82    public void tearDown() {
83    }
84  }
