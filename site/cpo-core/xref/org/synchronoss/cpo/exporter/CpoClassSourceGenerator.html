21  package org.synchronoss.cpo.exporter;
22  
23  import org.synchronoss.cpo.CpoException;
24  import org.synchronoss.cpo.MetaVisitor;
25  import org.synchronoss.cpo.meta.CpoMetaDescriptor;
26  import org.synchronoss.cpo.meta.domain.*;
27  
28  /**
29   * The CpoClassSourceGenerator generates java source code to define the cpo classes.
30   *
31   * @author Michael Bellomo
32   * @since 4/17/12
33   */
34  public class CpoClassSourceGenerator implements MetaVisitor {
35  
36    private static final String ATTR_PREFIX = "ATTR_";
37    private static final String FG_PREFIX = "FG_";
38    private static final String VAR_PREFIX = "_";
39    private static final String CLASS_SUFFIX = "Bean";
40  
41    protected CpoMetaDescriptor metaDescriptor;
42    protected String className = null;
43    protected StringBuilder header = new StringBuilder();
44    protected StringBuilder attributeStatics = new StringBuilder();
45    protected StringBuilder functionGroupStatics = new StringBuilder();
46    protected StringBuilder properties = new StringBuilder();
47    protected StringBuilder constructor = new StringBuilder();
48    protected StringBuilder gettersSetters = new StringBuilder();
49    protected StringBuilder equals = new StringBuilder();
50    protected StringBuilder hashCode = new StringBuilder();
51    protected StringBuilder toString = new StringBuilder();
52    protected StringBuilder footer = new StringBuilder();
53  
54    public CpoClassSourceGenerator(CpoMetaDescriptor metaDescriptor) {
55      this.metaDescriptor = metaDescriptor;
56    }
57  
58    /**
59     * Resets all of the buffers.
60     *
61     * This is needed if you intend to reuse the visitor. This is always called by visit(CpoClass)
62     */
63    protected void reset() {
64      attributeStatics = new StringBuilder();
65      functionGroupStatics = new StringBuilder();
66      properties = new StringBuilder();
67      constructor = new StringBuilder();
68      gettersSetters = new StringBuilder();
69      equals = new StringBuilder();
70      hashCode = new StringBuilder();
71      toString = new StringBuilder();
72      footer = new StringBuilder();
73    }
74  
75    public String getClassName() {
76      return className;
77    }
78  
79    public String getSourceCode() {
80      StringBuilder source = new StringBuilder();
81  
82      source.append("/* \n");
83      source.append(" *  This class auto-generated by " + this.getClass().getName() + "\n");
84      source.append(" */\n");
85      source.append(header);
86      source.append("\n");
87      source.append("  /* Properties */\n");
88      source.append(properties);
89      source.append("\n");
90      source.append("  /* Constructor */\n");
91      source.append(constructor);
92      source.append("\n");
93      source.append("  /* Getters and Setters */\n");
94      source.append(gettersSetters);
95  
96      // generate equals()
97      source.append("  public boolean equals(Object o) {\n");
98      source.append("    if (this == o)\n");
99      source.append("      return true;\n");
100     source.append("    if (o == null || getClass() != o.getClass())\n");
101     source.append("      return false;\n");
102     source.append("\n");
103     source.append("    " + className + " that = (" + className + ")o;\n");
104     source.append("\n");
105     source.append(equals);
106     source.append("\n");
107     source.append("    return true;\n");
108     source.append("  }\n\n");
109 
110     // generate hashCode()
111     source.append("  public int hashCode() {\n");
112     source.append("    int result = 0;\n");
113     source.append("    result = 31 * result + getClass().getName().hashCode();\n");
114     source.append(hashCode);
115     source.append("    return result;\n");
116     source.append("  }\n\n");
117 
118     // generate toString()
119     source.append("  public String toString() {\n");
120     source.append("    StringBuilder str = new StringBuilder();\n");
121     source.append(toString);
122     source.append("    return str.toString();\n");
123     source.append("  }\n");
124 
125     source.append(footer);
126     source.append("\n");
127 
128     return source.toString();
129   }
130 
131   /**
132    * The cpoClass name will be used for the name of the interface
133    */
134   protected String generateInterfaceName(CpoClass cpoClass) {
135     String className = cpoClass.getName();
136     if (className.lastIndexOf(".") != -1) {
137       className = className.substring(className.lastIndexOf(".") + 1);
138     }
139     return className;
140   }
141 
142   /**
143    * Returns the name of the class to use
144    */
145   protected String generateClassName(CpoClass cpoClass) {
146     return generateInterfaceName(cpoClass) + CLASS_SUFFIX;
147   }
148 
149   @Override
150   public void visit(CpoClass cpoClass) {
151 
152     // reset all the buffers
153     reset();
154 
155     className = generateClassName(cpoClass);
156     String interfaceName = generateInterfaceName(cpoClass);
157 
158     // generate class header
159     if (cpoClass.getName().lastIndexOf(".") != -1) {
160       String packageName = cpoClass.getName().substring(0, cpoClass.getName().lastIndexOf("."));
161       header.append("package " + packageName + ";\n\n");
162     }
163 
164     // generate class declaration
165     header.append("public class " + className + " implements " + interfaceName + ", java.io.Serializable {\n");
166 
167     // footer
168     footer.append("}\n");
169 
170     // generate constructor
171     constructor.append("  public " + className + "() {\n");
172     constructor.append("  }\n");
173   }
174 
175   @Override
176   public void visit(CpoAttribute cpoAttribute) {
177 
178     String attName = scrubName(cpoAttribute.getJavaName());
179 
180     if (cpoAttribute.getTransformClassName()!=null && cpoAttribute.getTransformInMethod()==null) {
181       try {
182         cpoAttribute.loadRunTimeInfo(metaDescriptor, null);
183       } catch (Exception e) {
184 
185       }
186     }
187 
188     // the getter name is get concatenated with the camel case of the attribute name
189     String getterName;
190     String setterName;
191     if (attName.length() > 1) {
192       getterName = ("get" + attName.substring(0, 1).toUpperCase() + attName.substring(1) + "()");
193       setterName = ("set" + attName.substring(0, 1).toUpperCase() + attName.substring(1));
194     } else {
195       getterName = ("get" + attName.toUpperCase() + "()");
196       setterName = ("set" + attName.toUpperCase());
197     }
198 
199     try {
200       Class<?> attClass = metaDescriptor.getDataTypeJavaClass(cpoAttribute);
201       String attClassName = metaDescriptor.getDataTypeName(cpoAttribute);
202 
203       // generate attribute statics
204       String staticName = ATTR_PREFIX + attName.toUpperCase();
205       attributeStatics.append("  public final static String " + staticName + " = \"" + attName + "\";\n");
206 
207       // generate property declarations
208       properties.append("  private " + attClassName + " " + VAR_PREFIX + attName + ";\n");
209 
210       // generate getter
211       gettersSetters.append("  public " + attClassName + " " + getterName + " {\n");
212       gettersSetters.append("    return " + VAR_PREFIX + attName + ";\n");
213       gettersSetters.append("  }\n\n");
214 
215       // generate setter
216       gettersSetters.append("  public void " + setterName + "(" + attClassName + " " + attName + ") {\n");
217       gettersSetters.append("    " + VAR_PREFIX + attName + " = " + attName + ";\n");
218       gettersSetters.append("  }\n\n");
219 
220       // equals()
221       if (attClass.isPrimitive()) {
222         // primitive type, use ==
223         equals.append("    if (" + getterName + " != that." + getterName + ")\n");
224       } else if (attClass.isArray()) {
225         // array type, use Array.equals()
226         equals.append("    if (!java.util.Arrays.equals(" + getterName + ", that." + getterName + "))\n");
227       } else {
228         // object, use .equals
229         equals.append("    if (" + getterName + " != null ? !" + getterName + ".equals(that." + getterName + ") : that." + getterName + " != null)\n");
230       }
231       equals.append("      return false;\n");
232 
233       // hashCode()
234       if (attClass.isPrimitive()) {
235         // primitive type, need some magic
236         hashCode.append("    result = 31 * result + (String.valueOf(" + getterName + ").hashCode());\n");
237       } else if (attClass.isArray()) {
238         // array type, use Array.hashCode()
239         hashCode.append("    result = 31 * result + (" + getterName + "!= null ? java.util.Arrays.hashCode(" + getterName + ") : 0);\n");
240       } else {
241         hashCode.append("    result = 31 * result + (" + getterName + " != null ? " + getterName + ".hashCode() : 0);\n");
242       }
243 
244       // toString()
245       toString.append("    str.append(\"" + attName + " = \" + " + getterName + " + \"\\n\");\n");
246     } catch(CpoException ce) {
247 
248     }
249   }
250 
251   @Override
252   public void visit(CpoFunctionGroup cpoFunctionGroup) {
253 
254     // generate statics for function group
255     String fgName = cpoFunctionGroup.getName();
256     if (fgName == null) {
257       fgName = "NULL";
258     }
259     fgName = scrubName(fgName);
260 
261     String staticName = FG_PREFIX + cpoFunctionGroup.getType() + "_" + fgName.toUpperCase();
262 
263     if (cpoFunctionGroup.getName() == null) {
264       functionGroupStatics.append("  public final static String " + staticName + " = null;\n");
265     } else {
266       functionGroupStatics.append("  public final static String " + staticName + " = \"" + fgName + "\";\n");
267     }
268   }
269 
270   @Override
271   public void visit(CpoFunction cpoFunction) {
272     // nothing to do
273   }
274 
275   @Override
276   public void visit(CpoArgument cpoArgument) {
277     // nothing to do
278   }
279 
280   protected String scrubName(String name) {
281     return name.replaceAll("[^0-9a-zA-Z_]", "_");
282   }
