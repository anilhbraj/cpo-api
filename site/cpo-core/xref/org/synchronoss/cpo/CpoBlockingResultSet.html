21  package org.synchronoss.cpo;
22  
23  import org.slf4j.*;
24  
25  import java.util.*;
26  import java.util.concurrent.LinkedBlockingQueue;
27  
28  public class CpoBlockingResultSet<E> implements CpoResultSet<E>, Iterator<E> {
29  
30    private static final Logger logger = LoggerFactory.getLogger(CpoBlockingResultSet.class);
31    private static final long serialVersionUID = 1L;
32    private int capacity = 0;
33    private final ThreadLocal<E> tlObj = new ThreadLocal<>();
34    private LinkedBlockingQueue<E> lbq = null;
35    private final Set<Thread> producers = new HashSet<>();
36    private final Set<Thread> consumers = new HashSet<>();
37  
38    private CpoBlockingResultSet() {
39    }
40  
41    public CpoBlockingResultSet(int capacity) {
42      this.capacity = capacity;
43      lbq = new LinkedBlockingQueue<>(capacity);
44    }
45  
46    @Override
47    public void put(E e) throws InterruptedException {
48      producers.add(Thread.currentThread());
49      logger.debug("Put Called");
50      lbq.put(e);
51    }
52  
53    @Override
54    public boolean hasNext() {
55      logger.debug("hasNext Called");
56  
57      if (tlObj.get() != null || lbq.size() > 0) {
58        return true;
59      }
60  
61      if (lbq.size() == 0 && Thread.interrupted()) {
62        return false;
63      }
64  
65      try {
66        tlObj.set(lbq.take());
67      } catch (InterruptedException ie) {
68        logger.error("CpoBlockingResultSet.hasNext() - Interrupted and bailing out");
69        return false;
70      }
71      return true;
72    }
73  
74    @Override
75    public int size() {
76      return lbq.size();
77    }
78  
79    @Override
80    public void remove() {
81      throw new UnsupportedOperationException();
82    }
83  
84    @Override
85    public E next() throws NoSuchElementException {
86      logger.debug("next Called");
87      E ret = tlObj.get();
88  
89      if (ret == null) {
90        if (lbq.size() == 0 && Thread.interrupted()) {
91          throw new NoSuchElementException();
92        }
93  
94        try {
95          ret = take();
96        } catch (InterruptedException ie) {
97          logger.error("CpoBlockingResultSet.next() - Interrupted and bailing out");
98          throw new NoSuchElementException();
99        }
100     } else {
101       tlObj.set(null);
102     }
103 
104     return ret;
105   }
106 
107   @Override
108   public Iterator<E> iterator() {
109     return this;
110   }
111 
112   @Override
113   public E take() throws InterruptedException {
114     consumers.add(Thread.currentThread());
115     logger.debug("Take Called");
116     return lbq.take();
117   }
118 
119   @Override
120   public void cancel() {
121     for (Thread t : consumers) {
122       t.interrupt();
123     }
124     for (Thread t : producers) {
125       t.interrupt();
126     }
127   }
128 
129   @Override
130   public int getFetchSize() {
131     return capacity;
132   }
133 }
