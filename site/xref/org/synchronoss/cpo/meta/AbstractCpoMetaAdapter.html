53    @Override
54    public <T> CpoClass getMetaClass(T obj) throws CpoException {
55      CpoClass cpoClass = null;
56      String className;
57      String requestedName;
58      Class<?> classObj;
59      Class<?> requestedClass;
60  
61      if (obj != null) {
62        requestedClass = obj.getClass();
63        classObj = requestedClass;
64        requestedName = requestedClass.getName();
65        className = requestedName;
66        cpoClass = classMap.get(className);
67  
68        while (cpoClass == null && classObj != null) {
69          classObj = classObj.getSuperclass();
70          className = classObj == null ? null : classObj.getName();
71          if (className != null) {
72            cpoClass = classMap.get(className);
73          }
74        }
75        if (cpoClass == null) {
76          throw new CpoException("No Metadata found for class:" + requestedName);
77        }
78      }
79  
80      return cpoClass;
81    }

92      String clazzName = getDataTypeJavaClass(attribute).getName();
93      byte[] b = new byte[0];
94      char[] c = new char[0];
95  
96      if (b.getClass().getName().equals(clazzName)) {

98      } else if (c.getClass().getName().equals(clazzName)) {

135   protected void loadCpoMetaDataDocument(CpoMetaDataDocument metaDataDoc, boolean caseSensitive) throws CpoException {
136     for (CtClass ctClass : metaDataDoc.getCpoMetaData().getCpoClassArray()) {
137 
138       CpoClass cpoClass = getCpoClass(ctClass.getName());
139       if (cpoClass == null) {
140         cpoClass = createCpoClass(caseSensitive);
141         cpoClass.setName(ctClass.getName());
142         cpoClass.setDescription(ctClass.getDescription());
143         loadCpoClass(cpoClass, ctClass);
144         addCpoClass(cpoClass);
145       } else {
146         loadCpoClass(cpoClass, ctClass);
147       }
148     }
149   }

151   protected void loadCpoClass(CpoClass cpoClass, CtClass ctClass) throws CpoException {
152     logger.debug("Loading class: " + ctClass.getName());
153 
154     currentClass = cpoClass;
155 
156     for (CtAttribute ctAttribute : ctClass.getCpoAttributeArray()) {
157       CpoAttribute cpoAttribute = cpoClass.getAttributeJava(ctAttribute.getJavaName());
158 
159       if (cpoAttribute == null) {
160         cpoAttribute = createCpoAttribute();
161         loadCpoAttribute(cpoAttribute, ctAttribute);
162         cpoClass.addAttribute(cpoAttribute);
163       } else {
164         loadCpoAttribute(cpoAttribute, ctAttribute);
165       }
166     }
167 
168     for (CtFunctionGroup ctFunctionGroup : ctClass.getCpoFunctionGroupArray()) {
169       CpoFunctionGroup functionGroup = null;
170 
171       try {
172         functionGroup = cpoClass.getFunctionGroup(ctFunctionGroup.getType().toString(), ctFunctionGroup.getName());
173       } catch (Exception e) {
174         // this a runtime exception that we can ignore during load time.
175         if (logger.isTraceEnabled()) {
176           logger.trace(e.getLocalizedMessage());
177         }
178       }
179 
180       if (functionGroup == null) {
181         functionGroup = createCpoFunctionGroup();
182         loadCpoFunctionGroup(functionGroup, ctFunctionGroup);
183         cpoClass.addFunctionGroup(functionGroup);
184       } else {
185         functionGroup.clearFunctions();
186         loadCpoFunctionGroup(functionGroup, ctFunctionGroup);
187       }
188       logger.debug("Added Function Group: " + functionGroup.getName());
189     }
190   }

201   protected void loadCpoFunctionGroup(CpoFunctionGroup cpoFunctionGroup, CtFunctionGroup ctFunctionGroup) {
202     cpoFunctionGroup.setDescription(ctFunctionGroup.getDescription());
203     if (ctFunctionGroup.isSetName()) {
204       cpoFunctionGroup.setName(ctFunctionGroup.getName());
205     }
206     cpoFunctionGroup.setType(ctFunctionGroup.getType().toString());
207 
208     for (CtFunction ctFunction : ctFunctionGroup.getCpoFunctionArray()) {
209       CpoFunction cpoFunction = createCpoFunction();
210       cpoFunctionGroup.addFunction(cpoFunction);
211       loadCpoFunction(cpoFunction, ctFunction);
212     }
213   }

215   protected void loadCpoFunction(CpoFunction cpoFunction, CtFunction ctFunction) {
216     cpoFunction.setName(ctFunction.getName());
217     cpoFunction.setExpression(ctFunction.getExpression());
218     cpoFunction.setDescription(ctFunction.getDescription());
219 
220     for (CtArgument ctArgument : ctFunction.getCpoArgumentArray()) {
221       CpoArgument cpoArgument = createCpoArgument();
222       cpoFunction.addArgument(cpoArgument);
223       loadCpoArgument(cpoArgument, ctArgument);

262   protected void addCpoClass(CpoClass metaClass) {
263     CpoClass oldMetaClass = classMap.put(metaClass.getName(), metaClass);
264     if (oldMetaClass != null) {
265       logger.debug("Overwrote class: " + metaClass.getName());
266     } else {
267       logger.debug("Added class: " + metaClass.getName());
268     }
269   }

271   protected void removeCpoClass(CpoClass metaClass) {
272     if (metaClass != null) {
273       logger.debug("Removing class: " + metaClass.getName());
274       metaClass.emptyMaps();
275       classMap.remove(metaClass.getName());
276     }
277   }
