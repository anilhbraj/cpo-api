21  package org.synchronoss.cpo;
22  
23  import org.synchronoss.cpo.meta.domain.CpoAttribute;
24  import org.synchronoss.cpo.meta.domain.CpoClass;
25  
26  import java.util.*;
27  
28  /**
29   * BindableWhereBuilder is an interface for specifying the sort order in which objects are returned from the Datasource.
30   *
31   * @author david berry
32   */
33  public class BindableWhereBuilder<T> implements NodeVisitor {
34  
35    private StringBuilder whereClause = new StringBuilder();
36    private CpoClass cpoClass = null;
37    private Collection<BindAttribute> bindValues = new ArrayList<>();
38  
39    public String getWhereClause() {
40      return whereClause.toString();
41    }
42  
43    public Collection<BindAttribute> getBindValues() {
44      return this.bindValues;
45    }
46  
47    @SuppressWarnings("unused")
48    private BindableWhereBuilder() {
49    }
50  
51    public BindableWhereBuilder(CpoClass cpoClass) {
52      this.cpoClass = cpoClass;
53    }
54  
55    /**
56     * This is called by composite nodes prior to visiting children
57     *
58     * @param node The node to be visited
59     * @return a boolean (false) to end visit or (true) to continue visiting
60     */
61    @Override
62    public boolean visitBegin(Node node) throws Exception {
63      BindableCpoWhere jcw = (BindableCpoWhere) node;
64      whereClause.append(jcw.toString(cpoClass));
65      if (jcw.hasParent() || jcw.getLogical() != CpoWhere.LOGIC_NONE) {
66        whereClause.append(" (");
67      } else {
68        whereClause.append(" ");
69      }
70  
71      return true;
72    }
73  
74    /**
75     * This is called for composite nodes between visiting children
76     *
77     * @param node The node to be visited
78     * @return a boolean (false) to end visit or (true) to continue visiting
79     */
80    @Override
81    public boolean visitMiddle(Node node) throws Exception {
82      return true;
83    }
84  
85    /**
86     * This is called by composite nodes after visiting children
87     *
88     * @param node The node to be visited
89     * @return a boolean (false) to end visit or (true) to continue visiting
90     */
91    @Override
92    public boolean visitEnd(Node node) throws Exception {
93      BindableCpoWhere bcw = (BindableCpoWhere) node;
94      if (bcw.hasParent() || bcw.getLogical() != CpoWhere.LOGIC_NONE) {
95        whereClause.append(")");
96      }
97      return true;
98    }
99  
100   /**
101    * This is called for component elements which have no children
102    *
103    * @param node The element to be visited
104    * @return a boolean (false) to end visit or (true) to continue visiting
105    */
106   @Override
107   public boolean visit(Node node) throws Exception {
108     BindableCpoWhere bcw = (BindableCpoWhere) node;
109     CpoAttribute attribute;
110     whereClause.append(bcw.toString(cpoClass));
111     if (bcw.getValue() != null) {
112       attribute = cpoClass.getAttributeJava(bcw.getAttribute());
113       if (attribute == null) {
114         attribute = cpoClass.getAttributeJava(bcw.getRightAttribute());
115       }
116       if (attribute == null) {
117         if (bcw.getComparison() == CpoWhere.COMP_IN && bcw.getValue() instanceof Collection) {
118           for (Object obj : (Collection) bcw.getValue()) {
119             bindValues.add(new BindAttribute(bcw.getAttribute() == null ? bcw.getRightAttribute() : bcw.getAttribute(), obj));
120           }
121         } else {
122           bindValues.add(new BindAttribute(bcw.getAttribute() == null ? bcw.getRightAttribute() : bcw.getAttribute(), bcw.getValue()));
123         }
124       } else {
125         if (bcw.getComparison() == CpoWhere.COMP_IN && bcw.getValue() instanceof Collection) {
126           for (Object obj : (Collection) bcw.getValue()) {
127             bindValues.add(new BindAttribute(attribute, obj));
128           }
129         } else {
130           bindValues.add(new BindAttribute(attribute, bcw.getValue()));
131         }
132       }
133     }
134     return true;
135   }
136 }
