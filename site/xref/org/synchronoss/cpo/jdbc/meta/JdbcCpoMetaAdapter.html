21  package org.synchronoss.cpo.jdbc.meta;
22  
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  import org.synchronoss.cpo.core.cpoCoreMeta.*;
26  import org.synchronoss.cpo.jdbc.*;
27  import org.synchronoss.cpo.jdbc.cpoJdbcMeta.*;
28  import org.synchronoss.cpo.parser.BoundExpressionParser;
29  import org.synchronoss.cpo.meta.AbstractCpoMetaAdapter;
30  import org.synchronoss.cpo.meta.DataTypeMapEntry;
31  import org.synchronoss.cpo.meta.DataTypeMapper;
32  import org.synchronoss.cpo.meta.domain.*;
33  import org.synchronoss.cpo.parser.ExpressionParser;
34  
35  import java.math.BigDecimal;
36  import java.util.*;
37  
38  /**
39   * @author dberry
40   */
41  public class JdbcCpoMetaAdapter extends AbstractCpoMetaAdapter {
42    private static final Logger logger = LoggerFactory.getLogger(JdbcCpoMetaAdapter.class);
43    private static final DataTypeMapEntry<String> defaultDataTypeMapEntry = new DataTypeMapEntry<>(java.sql.Types.VARCHAR, "VARCHAR", String.class); //12
44    private static final DataTypeMapper dataTypeMapper = initDataTypeMapper();
45  
46    @Override
47    protected DataTypeMapper getDataTypeMapper() {
48      return dataTypeMapper;
49    }
50  
51    @Override
52    protected void loadCpoAttribute(CpoAttribute cpoAttribute, CtAttribute ctAttribute) {
53      super.loadCpoAttribute(cpoAttribute, ctAttribute);
54  
55      // cast to the expected subclasses
56      JdbcCpoAttribute jdbcAttribute = (JdbcCpoAttribute)cpoAttribute;
57      CtJdbcAttribute ctJdbcAttribute = (CtJdbcAttribute)ctAttribute;
58  
59      jdbcAttribute.setDbTable(ctJdbcAttribute.getDbTable());
60      jdbcAttribute.setDbColumn(ctJdbcAttribute.getDbColumn());
61    }
62  
63    @Override
64    protected void loadCpoArgument(CpoArgument cpoArgument, CtArgument ctArgument) {
65      super.loadCpoArgument(cpoArgument, ctArgument);
66  
67      // cast to the expected subclasses
68      JdbcCpoArgument jdbcArgument = (JdbcCpoArgument)cpoArgument;
69      CtJdbcArgument ctJdbcArgument = (CtJdbcArgument)ctArgument;
70  
71  //    logger.debug("Setting argument scope to: "+ctJdbcArgument.getScope().toString());
72      jdbcArgument.setScope(ctJdbcArgument.getScope().toString());
73      jdbcArgument.setTypeInfo(ctJdbcArgument.getTypeInfo());
74    }
75  
76    @Override
77    protected CpoAttribute createCpoAttribute() {
78      return new JdbcCpoAttribute();
79    }
80  
81    @Override
82    protected CpoArgument createCpoArgument() {
83      return new JdbcCpoArgument();
84    }
85  
86    @Override
87    public ExpressionParser getExpressionParser() {
88      return new BoundExpressionParser();
89    }
90  
91    @Override
92    public List<String> getAllowableDataTypes() {
93      return dataTypeMapper.getDataTypeNames();
94    }
95  
96    private static DataTypeMapper initDataTypeMapper() {
97      logger.debug("Initializing the DataMapper");
98      DataTypeMapper dataTypeMapper = new DataTypeMapper(defaultDataTypeMapEntry);
99  
100     // JDK 1.4.2 Values
101     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.CHAR, "CHAR", String.class)); // 1
102     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.LONGVARCHAR, "LONGVARCHAR", String.class)); // -1
103     dataTypeMapper.addDataTypeEntry(defaultDataTypeMapEntry); // 12
104     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.DECIMAL, "DECIMAL", BigDecimal.class)); // 3
105     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.NUMERIC, "NUMERIC", BigDecimal.class)); // 2
106     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.TINYINT, "TINYINT", byte.class)); // -6
107     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.SMALLINT, "SMALLINT", short.class)); // 5
108     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.INTEGER, "INTEGER", int.class)); // 4
109     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.BIGINT, "BIGINT", long.class)); // -5
110     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.REAL, "REAL", float.class)); // 7
111     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.FLOAT, "FLOAT", double.class)); // 6
112     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.DOUBLE, "DOUBLE", double.class)); // 8
113     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.BINARY, "BINARY", byte[].class)); // -2
114     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.VARBINARY, "VARBINARY", byte[].class)); // -3
115     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.LONGVARBINARY, "LONGVARBINARY", byte[].class)); // -4
116     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.DATE, "DATE", java.sql.Date.class)); // 91
117     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.TIME, "TIME", java.sql.Time.class)); // 92
118     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.TIMESTAMP, "TIMESTAMP", java.sql.Timestamp.class)); // 93
119     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.CLOB, "CLOB", java.sql.Clob.class)); // 2005
120     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.BLOB, "BLOB", java.sql.Blob.class)); // 2004
121     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.ARRAY, "ARRAY", java.sql.Array.class)); // 2003
122     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.REF, "REF", java.sql.Ref.class)); // 2006
123     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.DISTINCT, "DISTINCT", Object.class)); // 2001
124     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.STRUCT, "STRUCT", Object.class)); // 2002
125     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.OTHER, "OTHER", Object.class)); // 1111
126     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.JAVA_OBJECT, "JAVA_OBJECT", Object.class)); // 2000
127     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.DATALINK, "DATALINK", java.net.URL.class)); // 70
128     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.BIT, "BIT", boolean.class)); // -7
129     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(java.sql.Types.BOOLEAN, "BOOLEAN", boolean.class)); //16
130 
131     // dbspecific types needed to generate the class from a function.
132     dataTypeMapper.addDataTypeEntry(new DataTypeMapEntry<>(100, "VARCHAR_IGNORECASE", java.lang.String.class)); // HSQLDB TYPE for VARCHAR_IGNORE_CASE
133 
134     logger.debug("Returning the DataMapper");
135     return dataTypeMapper;
136   }
