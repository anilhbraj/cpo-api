21  package org.synchronoss.cpo.jdbc.config;
22  
23  import org.slf4j.*;
24  import org.synchronoss.cpo.*;
25  import org.synchronoss.cpo.config.CpoConfigProcessor;
26  import org.synchronoss.cpo.core.cpoCoreConfig.CtDataSourceConfig;
27  import org.synchronoss.cpo.jdbc.*;
28  import org.synchronoss.cpo.jdbc.cpoJdbcConfig.*;
29  import org.synchronoss.cpo.jdbc.meta.JdbcCpoMetaDescriptor;
30  import org.synchronoss.cpo.meta.CpoMetaDescriptor;
31  
32  import java.util.*;
33  
34  /**
35   *
36   * @author dberry
37   */
38  public class JdbcCpoConfigProcessor implements CpoConfigProcessor {
39  
40    private static final Logger logger = LoggerFactory.getLogger(JdbcCpoConfigProcessor.class);
41    private static final String PROP_URL1 = "url";
42    private static final String PROP_URL2 = "URL";
43    private static final String PROP_USER = "user";
44    private static final String PROP_PASSWORD = "password";
45  
46    public JdbcCpoConfigProcessor() {
47    }
48  
49    @Override
50    public CpoAdapter processCpoConfig(CtDataSourceConfig cpoConfig) throws CpoException {
51  
52      CpoAdapter cpoAdapter = null;
53  
54      if (cpoConfig == null || !(cpoConfig instanceof CtJdbcConfig)) {
55        throw new CpoException("Invalid Jdbc Configuration Information");
56      }
57  
58      CtJdbcConfig jdbcConfig = (CtJdbcConfig) cpoConfig;
59  
60      JdbcCpoMetaDescriptor metaDescriptor = (JdbcCpoMetaDescriptor)CpoMetaDescriptor.getInstance(jdbcConfig.getMetaDescriptorName());
61  
62      if (jdbcConfig.isSetSupportsBlobs())
63        metaDescriptor.setSupportsBlobs(jdbcConfig.getSupportsBlobs());
64  
65      if (jdbcConfig.isSetSupportsCalls())
66        metaDescriptor.setSupportsCalls(jdbcConfig.getSupportsCalls());
67  
68      if (jdbcConfig.isSetSupportsMillis())
69        metaDescriptor.setSupportsMillis(jdbcConfig.getSupportsMillis());
70  
71      if (jdbcConfig.isSetSupportsSelect4Update())
72        metaDescriptor.setSupportsSelect4Update(jdbcConfig.getSupportsSelect4Update());
73  
74      // build a datasource info
75      if (jdbcConfig.isSetReadWriteConfig()) {
76        DataSourceInfo dataSourceInfo = buildDataSourceInfo(jdbcConfig.getReadWriteConfig());
77        cpoAdapter = JdbcCpoAdapter.getInstance(metaDescriptor, dataSourceInfo);
78      } else {
79        DataSourceInfo readDataSourceInfo = buildDataSourceInfo(jdbcConfig.getReadConfig());
80        DataSourceInfo writeDataSourceInfo = buildDataSourceInfo(jdbcConfig.getWriteConfig());
81        cpoAdapter = JdbcCpoAdapter.getInstance(metaDescriptor, writeDataSourceInfo, readDataSourceInfo);
82      }
83  
84      return cpoAdapter;
85    }
86  
87    private DataSourceInfo buildDataSourceInfo(CtJdbcReadWriteConfig readWriteConfig) throws CpoException {
88      DataSourceInfo dataSourceInfo = null;
89  
90      if (readWriteConfig.isSetJndiName()) {
91        dataSourceInfo = new JndiJdbcDataSourceInfo(readWriteConfig.getJndiName());
92      } else if (readWriteConfig.isSetDataSourceClassName()) {
93        SortedMap<String, String> props = new TreeMap<>();
94  
95        if (readWriteConfig.isSetUrl()) {
96          props.put(PROP_URL1, readWriteConfig.getUrl());
97          props.put(PROP_URL2, readWriteConfig.getUrl());
98        }
99  
100       if (readWriteConfig.isSetUser()) {
101         props.put(PROP_USER, readWriteConfig.getUser());
102       }
103 
104       if (readWriteConfig.isSetPassword()) {
105         props.put(PROP_PASSWORD, readWriteConfig.getPassword());
106       }
107 
108       for (CtProperty property : readWriteConfig.getPropertyArray()){
109         props.put(property.getName(), property.getValue());
110         logger.debug("Adding property "+property.getName()+"="+property.getValue());
111       }
112 
113       dataSourceInfo = new ClassJdbcDataSourceInfo(readWriteConfig.getDataSourceClassName(), props);
114     } else if (readWriteConfig.isSetDriverClassName()) {
115       if (readWriteConfig.isSetUser()) {
116         dataSourceInfo = new DriverJdbcDataSourceInfo(readWriteConfig.getDriverClassName(), readWriteConfig.getUrl(), readWriteConfig.getUser(), readWriteConfig.getPassword());
117       } else if (readWriteConfig.getPropertyArray().length>0) {
118         Properties props = new Properties();
119         for (CtProperty property : readWriteConfig.getPropertyArray()){
120           props.put(property.getName(), property.getValue());
121         }
122         dataSourceInfo = new DriverJdbcDataSourceInfo(readWriteConfig.getDriverClassName(), readWriteConfig.getUrl(), props);
123       } else {
124         dataSourceInfo = new DriverJdbcDataSourceInfo(readWriteConfig.getDriverClassName(), readWriteConfig.getUrl());
125       }
126     }
127 
128     logger.debug("Created DataSourceInfo: "+dataSourceInfo);
129     return dataSourceInfo;
130   }
